public class HandEvaluator {
    public List<HandResult> EvaluateHand(int[] dice) {
        List<HandResult> results = new List<HandResult>();
        
        // ì •ë ¬
        int[] sorted = dice.OrderBy(d => d).ToArray();
        
        // ê° ì¡±ë³´ ì²´í¬
        if (IsYahtzee(sorted)) 
            results.Add(new HandResult(HandType.Yahtzee, 50, 10f));
        if (IsLargeStraight(sorted)) 
            results.Add(new HandResult(HandType.LargeStraight, 40, 5f));
        if (IsSmallStraight(sorted)) 
            results.Add(new HandResult(HandType.SmallStraight, 30, 4f));
        if (IsFullHouse(sorted)) 
            results.Add(new HandResult(HandType.FullHouse, 25, 3f));
        if (IsFourOfAKind(sorted)) 
            results.Add(new HandResult(HandType.FourOfAKind, Sum(dice), 4f));
        if (IsThreeOfAKind(sorted)) 
            results.Add(new HandResult(HandType.ThreeOfAKind, Sum(dice), 2f));
        
        // í•­ìƒ ì°¬ìŠ¤ ê°€ëŠ¥
        results.Add(new HandResult(HandType.Chance, Sum(dice), 1f));
        
        // ìˆ«ìë³„ (ì›ìŠ¤~ì‹ìŠ¤)
        for (int i = 1; i <= 6; i++) {
            int count = dice.Count(d => d == i);
            if (count > 0) {
                results.Add(new HandResult(
                    (HandType)i, 
                    i * count, 
                    1f
                ));
            }
        }
        
        return results;
    }
    
    private bool IsYahtzee(int[] sorted) {
        return sorted.Distinct().Count() == 1;
    }
    
    private bool IsLargeStraight(int[] sorted) {
        return sorted.SequenceEqual(new[] {1,2,3,4,5}) 
            || sorted.SequenceEqual(new[] {2,3,4,5,6});
    }
    
    private bool IsSmallStraight(int[] sorted) {
        var unique = sorted.Distinct().ToArray();
        return ContainsSequence(unique, new[] {1,2,3,4})
            || ContainsSequence(unique, new[] {2,3,4,5})
            || ContainsSequence(unique, new[] {3,4,5,6});
    }
    
    private bool IsFullHouse(int[] sorted) {
        var groups = sorted.GroupBy(d => d).ToArray();
        return groups.Length == 2 
            && groups.Any(g => g.Count() == 3) 
            && groups.Any(g => g.Count() == 2);
    }
}
